# ជ្រើសរើសគ្រឿងផ្សំថ្មីៗ និងស្អាតបាត

ពេលមានភ្ញៀវមកពិសាអាហារ ម្ចាស់កម្មវិធីតែងតែបារម្ភពីសុខភាពរបស់ចាស់ទុំ និងភ្ញៀវទាំងអស់។ ការញ៉ាំម្ហូបដែលស្រស់ និងស្អាត ជារឿងដែលយើងយកចិត្តទុកដាក់បំផុត។

## ការជ្រើសរើសដោយផ្ទាល់ដៃ
រៀងរាល់ព្រឹកព្រលឹម ក្រុមការងារយើងចុះទៅរើសសាច់ និងគ្រឿងសមុទ្រដោយផ្ទាល់។ យើងមើលរាល់ចំណុច ដើម្បីប្រាកដថាអ្វីៗទាំងអស់គឺស្រស់ល្អ មុននឹងយកមកចម្អិន។

## ការលាងសម្អាតយ៉ាងយកចិត្តទុកដាក់
យើងចាត់ទុកបន្លែ និងសាច់ដូចជាធ្វើញ៉ាំនៅក្នុងផ្ទះខ្លួនឯង។ យើងលាងទឹកច្រើនដងរហូតដល់អស់ដី និងភាពកខ្វក់ ទើបយកទៅទុកក្នុងធុងទឹកកកយ៉ាងរៀបរយ។


## ថែរក្សាភាពស្រស់រហូតដល់ពេលញ៉ាំ
យើងមិនទុកសាច់ចោលនៅខាងក្រៅទេ។ អ្វីៗត្រូវបានក្លាសេក្នុងសីតុណ្ហភាពត្រឹមត្រូវ ដើម្បីឱ្យពេលយកមកចម្អិន វានៅតែមានរសជាតិផ្អែមធម្មជាតិ។

### បើមានក្តីបារម្ភរឿងម្ហូបអាហារ អាចប្រាប់យើងបាន
រឿងម្ហូបស្អាតបាត និងស្រស់ល្អ គឺជារឿងដែលយើងធ្វើជាប្រចាំ។ បើមានអ្វីចង់ផ្តាំផ្ញើ អាចប្រាប់យើងបានដោយក្តីទុកចិត្ត។
