# រក្សាបរិយាកាសឱ្យនៅកក់ក្តៅជានិច្ច

ពេលរៀបចំកម្មវិធីនៅខាងក្រៅ មនុស្សភាគច្រើនតែងតែបារម្ភពេលមើលមេឃ។ ប៉ុន្តែមិនថាអាកាសធាតុប្រែប្រួលយ៉ាងណាទេ ភ្ញៀវគួរតែបានអង្គុយទទួលទានអាហារដោយក្តីសុខសាន្ត។

## ការពារពីខ្យល់និងភ្លៀង
ពេលមានភ្លៀងធ្លាក់ ឬខ្យល់បក់ យើងនឹងទម្លាក់ផ្ទាំងបាំងចុះយ៉ាងស្ងៀមស្ងាត់។ វាមិនមានសំឡេងរំខានទេ ហើយភ្ញៀវនៅខាងក្នុងនៅតែអាចបន្តជជែកគ្នាបានជាធម្មតា។

## រក្សាពន្លឺឱ្យនៅភ្លឺជានិច្ច
បញ្ហាដាច់ភ្លើងជារឿងអាចកើតឡើង។ យើងបានរៀបចំប្រព័ន្ធទុកជាមុនរួចហើយ។ ដូច្នេះទោះបីជាភ្លើងដាច់ភ្លាមៗ ក៏អំពូល និងកង្ហារនៅតែបន្តដំណើរការដូចដើម ដោយស្ទើរតែគ្មានអ្នកចាប់អារម្មណ៍ដឹងឡើយ។


## មើលថែភាពស្អាតនៃផ្លូវដើរ
ពេលមានភ្លៀង ផ្លូវដើរអាចមានទឹកដក់។ ក្រុមការងារយើងនឹងតាមជូតសម្អាតចេញយ៉ាងលឿន និងស្ងាត់ៗ ដើម្បីឱ្យភ្ញៀវដើរទៅមកមិនប្រឡាក់សម្លៀកបំពាក់។

### បើមានការព្រួយបារម្ភ អាចប្រាប់យើងបាន
រឿងបរិយាកាស និងទីកន្លែង ទុកឱ្យក្រុមការងារយើងជាអ្នកមើលថែចុះ。
