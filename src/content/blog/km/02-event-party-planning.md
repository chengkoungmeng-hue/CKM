**រៀបរៀងដោយ៖ ទិន្នន័យវិភាគប្រតិបត្តិការ CKM**

នៅក្នុងការរៀបចំប្រតិបត្តិការភោជនាហារដែលមានទំហំចាប់ពី ៥០ ដល់ ២០០០ នាក់ បញ្ហាប្រឈមមិនមែនជាការចម្អិនអាហារនោះទេ ប៉ុន្តែវាគឺជាការគ្រប់គ្រងទែរម៉ូឌីណាមិក (Thermodynamics) និងភស្តុភារកម្មនៅក្នុងលំហ។ ការបម្រើប្រូតេអ៊ីនរាប់ពាន់ចំណែកក្នុងពេលតែមួយ ទាមទារឱ្យមានការធ្វើសមកាលកម្មយ៉ាងច្បាស់លាស់រវាងពេលវេលា សីតុណ្ហភាពបរិយាកាស និងចម្ងាយនៃការបញ្ជូន។ ប្រតិបត្តិការខ្នាតធំរបស់ CKM ត្រូវបានកសាងឡើងលើមូលដ្ឋាននៃវិស្វកម្មប្រព័ន្ធ ដោយបំប្លែងអថេរនៃបរិយាកាសឱ្យទៅជាទិន្នន័យដែលអាចគ្រប់គ្រងបាន។

## ១. ហេដ្ឋារចនាសម្ព័ន្ធអគ្គិសនី និងការវាយតម្លៃលំហ

បុរេលក្ខខណ្ឌនៃប្រតិបត្តិការខ្នាតធំ គឺការគ្រប់គ្រងថាមពលនិងលំហ។
មុនពេលប្រតិបត្តិការ វិស្វកររបស់ CKM ធ្វើការវាយតម្លៃសមត្ថភាពផ្ទុកចរន្តអគ្គិសនី (Electrical load capacity) នៃទីតាំងផ្ទាល់ ដើម្បីធានាថាវាអាចទ្រទ្រង់ម៉ាស៊ីនរក្សាសីតុណ្ហភាព និងប្រព័ន្ធខ្យល់ចេញចូលដោយគ្មានការធ្លាក់ចុះវ៉ុល (Voltage drop)។ រាល់ហានិភ័យនៃកង្វះថាមពលដែលនាំឱ្យប្រព័ន្ធខ្សែសង្វាក់ត្រជាក់ដួលរលំ ត្រូវបានលុបបំបាត់តាមរយៈការរៀបចំប្រព័ន្ធថាមពលបម្រុងកម្រិតឧស្សាហកម្ម លុះត្រាតែប៉ារ៉ាម៉ែត្រទាំងអស់ត្រូវបានផ្ទៀងផ្ទាត់ ទើបការបញ្ជូនអាហារអាចចាប់ផ្តើមបាន។

## ២. រចនាសម្ព័ន្ធម៉ឺនុយផ្អែកលើរូបវិទ្យាកម្ដៅ

ម៉ឺនុយសម្រាប់ប្រតិបត្តិការខ្នាតធំ ត្រូវតែរចនាឡើងដើម្បីទប់ទល់នឹងច្បាប់ទីពីរនៃទែរម៉ូឌីណាមិក។
មុខម្ហូបត្រូវបានបែងចែកតាមសន្ទស្សន៍នៃការបាត់បង់កម្តៅ និងការរក្សាសំណើម៖ អាហារបំពងត្រូវបានចាត់ចែងក្នុងគន្លងបញ្ជូនល្បឿនលឿន ដើម្បីទប់ស្កាត់ការស្រូបយកសំណើម (Moisture absorption) ចំណែកឯស៊ុបត្រូវបានរក្សាក្នុងប្រព័ន្ធទឹកកម្តៅថេរ (Water bath) ដែលមានចរន្តកម្ដៅបិទជិត។ ម៉ូដែលរៀបចំនេះ ធានាថាទម្រង់រូបវន្តនៃអាហារនៅតុទី១ និងតុទី២០០ មានភាពស្មើគ្នាបេះបិទ ទោះបីជាមានការពន្យារពេលនៃកម្មវិធីក៏ដោយ។

## ៣. ការបែងចែកធនធានមនុស្ស និងការគ្រប់គ្រងហានិភ័យ

សេវាកម្មខ្នាតធំ គឺជាការអនុវត្តអត្រាសមាមាត្រ (Ratio) និងគន្លងចរាចរណ៍យ៉ាងតឹងរ៉ឹង។
CKM គណនាចំនួនបុគ្គលិកដោយផ្អែកលើចម្ងាយបញ្ជូន និងទំហំនៃតុភ្ញៀវ ដើម្បីលុបបំបាត់ការកកស្ទះនៅចំណុចប្រសព្វ (Bottlenecks)។ សម្រាប់កម្មវិធីខ្នាតធំ យើងបំបែកតំបន់ប្រតិបត្តិការជាកូដពណ៌ និងប្រើប្រាស់ប្រព័ន្ធទាក់ទងវិទ្យុដើម្បីធ្វើសមកាលកម្មចលនាបម្រើ។ ក្នុងករណីមានអថេរខាងក្រៅ ដូចជាបម្រែបម្រួលអាកាសធាតុ ពិធីសារបម្រុង (Contingency protocols) នឹងត្រូវបានដំណើរការភ្លាមៗ ដើម្បីរក្សាស្ថិរភាពប្រតិបត្តិការដោយមិនមានការរំខានដល់ទិន្នផលចុងក្រោយ។

> ស្ថិរភាពនៃប្រតិបត្តិការខ្នាតធំ គឺជាអនុគមន៍នៃការធ្វើផែនការភស្តុភារកម្មយ៉ាងជាក់លាក់ និងការលុបបំបាត់អថេរ (Variables) ជាមុន មិនមែនជាការដោះស្រាយបញ្ហាតាមស្ថានភាពជាក់ស្តែងនោះទេ។

### ទិន្នន័យសម្រាប់ប្រតិបត្តិការក្រៅទីតាំង និងការពង្រីកមាត្រដ្ឋាន

**តើមានយន្តការអ្វីខ្លះសម្រាប់ទីតាំងក្រៅអាគារ ឬវីឡាឯកជនដែលគ្មានហេដ្ឋារចនាសម្ព័ន្ធផ្ទះបាយពាណិជ្ជកម្ម?**
សម្រាប់ទីតាំងដែលគ្មានរចនាសម្ព័ន្ធផ្ទះបាយពាណិជ្ជកម្ម CKM អនុវត្តការចល័តប្រព័ន្ធផ្ទះបាយម៉ូឌុល (Modular field kitchens)។ យើងរៀបចំប្រព័ន្ធផ្គត់ផ្គង់ថាមពលអគ្គិសនីឯករាជ្យ ប្រព័ន្ធចម្រោះទឹកស្អាត និងប្រព័ន្ធបញ្ចេញខ្យល់ (Exhaust systems) នៅនឹងកន្លែង។ នេះអនុញ្ញាតឱ្យការកែច្នៃអាហារមានលក្ខណៈរូបវន្តដូចគ្នានឹងការចម្អិននៅផ្ទះបាយកណ្តាល ធានាបាននូវការគ្រប់គ្រងកម្តៅ និងអនាម័យ១០០%។

**តើប្រព័ន្ធផ្គត់ផ្គង់មានប្រតិកម្មយ៉ាងណា ប្រសិនបើមានកំណើនភ្ញៀវលើសពីការគណនាភ្លាមៗ?**
ទិន្នន័យនៃការចូលរួមតែងតែមានគម្លាត។ ប្រព័ន្ធភស្តុភារកម្មរបស់ CKM រួមបញ្ចូលនូវស្តុកវត្ថុធាតុដើមបម្រុងទុក (Buffer inventory) ដែលត្រូវបានថែរក្សាក្នុងប្រព័ន្ធសីតុណ្ហភាពអសកម្ម (Passive cooling) នៅនឹងកន្លែង។ នៅពេលមានការកើនឡើងទិន្នន័យភ្ញៀវ ពិធីសារបញ្ចេញធនធានបន្ទាន់នឹងត្រូវបានធ្វើសកម្មភាព ដោយអនុញ្ញាតឱ្យមានការកែច្នៃបន្ថែមដោយមិនកាត់បន្ថយប៉ារ៉ាម៉ែត្រគុណភាព ឬបង្កឱ្យមានភាពយឺតយ៉ាវដល់ខ្សែសង្វាក់បញ្ជូនឡើយ។
