**រៀបរៀងដោយ៖ ទិន្នន័យវិភាគប្រតិបត្តិការ CKM**

នៅក្នុងប្រតិបត្តិការភោជនាហារខ្នាតធំ សមត្ថភាពរបស់ប្រព័ន្ធចម្អិនអាហារមិនត្រូវបានវាស់វែងដោយ "ភាពឆ្ងាញ់" នៃចានគំរូនោះទេ ប៉ុន្តែវាស់វែងដោយ "អត្រានៃការចម្លង (Replication rate)" ដ៏ត្រឹមត្រូវសម្រាប់ប្រូតេអ៊ីនរាប់ពាន់ចំណែក។ CKM បដិសេធរាល់អថេរនៃអារម្មណ៍ដោយអត្តនោម័ត ជំនួសមកវិញនូវការកសាងម៉ាទ្រីសគ្រប់គ្រងវត្ថុធាតុដើម និងស្តង់ដារជីវគីមីយ៉ាងតឹងរ៉ឹងបំផុត ដើម្បីលុបបំបាត់គម្លាតគុណភាព។

## ១. ភស្តុភារកម្មខ្សែសង្វាក់ត្រជាក់ និងការអភិរក្សកោសិកា

ការរក្សា "ភាពស្រស់" នៅក្នុងបរិបទវិទ្យាសាស្ត្រអាហារ គឺការពន្យារប្រតិកម្មអុកស៊ីតកម្ម និងការរិចរិលនៃកោសិកា (Cellular degradation)។
ប្រព័ន្ធផ្គត់ផ្គង់របស់ CKM ភ្ជាប់ផ្ទាល់ទៅនឹងប្រភពដើម ដោយអនុវត្តខ្សែសង្វាក់ត្រជាក់ (Cold Chain) ប្រកបដោយប៉ារ៉ាម៉ែត្រសីតុណ្ហភាពស្នូល 4°C យ៉ាងតឹងរ៉ឹង។ ការគ្រប់គ្រងទែរម៉ូឌីណាមិកនេះ ទប់ស្កាត់ការបាត់បង់សំណើម និងរក្សារចនាសម្ព័ន្ធរូបវន្តនៃប្រូតេអ៊ីន មុនពេលវាត្រូវបានបញ្ចូលទៅក្នុងខ្សែសង្វាក់កែច្នៃរបស់យើង។ កិច្ចសន្យាផ្គត់ផ្គង់ច្បាស់លាស់ លុបបំបាត់អថេរនៃទីផ្សារ ធានាបាននូវស្ថិរភាពទិន្នផលដោយមិនគិតពីកាលៈទេសៈខាងក្រៅ។

## ២. ពិធីសារសុវត្ថិភាពជីវសាស្រ្តអាហារ

សុវត្ថិភាពចំណីអាហារ គឺជាប្រព័ន្ធការពារហានិភ័យជីវសាស្រ្តដាច់ខាត។
នៅក្នុងមជ្ឈមណ្ឌលកែច្នៃរបស់ CKM យើងអនុវត្តការបំបែកគន្លងចរាចរណ៍រូបវន្ត (Physical traffic segregation) រវាងប្រូតេអ៊ីនឆៅ និងអាហារឆ្អិន ដើម្បីលុបបំបាត់ការឆ្លងបាក់តេរីខ្វែង (Cross-contamination) ឱ្យនៅកម្រិតសូន្យ។ រាល់ដំណើរការកែច្នៃកម្ដៅទាំងអស់ ត្រូវតែគោរពតាមស្តង់ដារចំណុចត្រួតពិនិត្យសំខាន់ (Critical Control Point - CCP) ដោយការផ្ទៀងផ្ទាត់សីតុណ្ហភាពស្នូលចុងក្រោយ។ ការគ្រប់គ្រងភ្នាក់ងារបង្កជំងឺ គឺជាតក្កវិជ្ជាវិស្វកម្មដែលគ្រប់គ្រងលើរាល់ចលនានៅក្នុងផ្ទះបាយ។

## ៣. យន្តការចម្លងរចនាសម្ព័ន្ធរសជាតិ

បញ្ហាប្រឈមធំបំផុតនៅក្នុងប្រតិបត្តិការខ្នាតធំ គឺគម្លាតរសជាតិ (Flavor deviation) រវាងទិន្នផលដំបូង និងចុងក្រោយ។
CKM ដោះស្រាយបញ្ហានេះដោយបំប្លែងរូបមន្តប្រពៃណីទៅជា "ម៉ាទ្រីសប៉ារ៉ាម៉ែត្រ" ដ៏ជាក់លាក់។ ម៉ាស់នៃគ្រឿងទេសត្រូវបានវាស់វែងជាក្រាម ហើយរង្វាស់នៃកម្តៅ និងពេលវេលាត្រូវបានគ្រប់គ្រងជាវិនាទី។ ការធ្វើស្តង់ដារូបនីយកម្មនេះ (Standardization) លុបបំបាត់កំហុសដោយមនុស្ស (Human errors) អនុញ្ញាតឱ្យមានការចម្លងទម្រង់រសជាតិ (Flavor profile) ដូចគ្នាបេះបិទសម្រាប់រាល់ចំណែកនៃអាហារដែលបានបម្រើ។

> ប្រតិបត្តិការចម្អិនអាហារខ្នាតធំ មិនមែនជាសិល្បៈនៃអារម្មណ៍នោះទេ ប៉ុន្តែវាគឺជាការអនុវត្តវិស្វកម្មជីវគីមីប្រកបដោយភាពច្បាស់លាស់ ដើម្បីលុបបំបាត់អេនត្រូពី (Entropy) និងបង្កើតលទ្ធផលដែលមិនប្រែប្រួល។

### ទិន្នន័យតម្រូវសម្រាប់ការគ្រប់គ្រងវត្ថុធាតុដើម និងគុណភាព

**តើប្រព័ន្ធចម្អិនអាចឆ្លើយតបនឹងការផ្លាស់ប្តូររចនាសម្ព័ន្ធរសជាតិភ្លាមៗនៅថ្ងៃប្រតិបត្តិការបានដែរឬទេ?**
ប្រព័ន្ធកែច្នៃរបស់ CKM ដំណើរការផ្អែកលើម៉ាទ្រីសរសជាតិដែលបានធ្វើសមកាលកម្មទុកជាមុន។ ការកែប្រែប៉ារ៉ាម៉ែត្រទ្រង់ទ្រាយធំភ្លាមៗ នឹងរំខានដល់ប្រតិបត្តិការខ្សែសង្វាក់ និងបង្កើតគម្លាតទិន្នផល។ ហេតុនេះ ការកំណត់ប៉ារ៉ាម៉ែត្រត្រូវបញ្ចប់នៅដំណាក់កាលធ្វើតេស្ត។ ទោះជាយ៉ាងណា សម្រាប់តម្រូវការជីវសាស្រ្តជាក់លាក់ (ការទប់ស្កាត់អាឡែហ្ស៊ី ឬរបបអាហារបួស) ប្រព័ន្ធមានខ្សែសង្វាក់កែច្នៃឯករាជ្យបម្រុងទុក ដើម្បីគ្រប់គ្រងអថេរតូចតាចទាំងនេះដោយមិនរំខានដល់លំហូរទិន្នផលចម្បង។

**តើមានគម្លាតនៃស្តង់ដារប្រតិបត្តិការរវាងកម្មវិធីខ្នាតតូច និងខ្នាតធំដែរឬទេ?**
ប៉ារ៉ាម៉ែត្រនៃសុវត្ថិភាពជីវសាស្រ្ត និងវត្ថុធាតុដើម ត្រូវបានអនុវត្តដោយស្តង់ដារតែមួយគត់ដោយគ្មានគម្លាត។ អថេរតែមួយគត់គឺ "ទម្រង់នៃការបញ្ជូន (Delivery format)"។ កម្មវិធីខ្នាតតូចអនុញ្ញាតឱ្យមានការកែច្នៃដំណាក់កាលចុងក្រោយនៅក្បែរតុ (Tableside processing) ខណៈដែលប្រតិបត្តិការខ្នាតធំតម្រូវឱ្យមានការធ្វើសមកាលកម្មកម្តៅ និងចលនាបញ្ជូនយ៉ាងតឹងរ៉ឹងបំផុតពីផ្ទះបាយកណ្តាល ដើម្បីរក្សាទែរម៉ូឌីណាមិកនៃអាហាររហូតដល់តុភ្ញៀវ។
