**រៀបរៀងដោយ៖ ទិន្នន័យវិភាគប្រតិបត្តិការ CKM**

នៅក្នុងប្រតិបត្តិការភោជនាហារខ្នាតធំ ហានិភ័យមិនមែនកើតចេញពីគុណភាពអាហារនោះទេ ប៉ុន្តែកើតចេញពីការដួលរលំនៃហេដ្ឋារចនាសម្ព័ន្ធទីតាំង និងការដាច់ចង្វាក់ភស្តុភារកម្ម។ ទិន្នផលរសជាតិដ៏ល្អឥតខ្ចោះនៅក្នុងដំណាក់កាលធ្វើតេស្ត នឹងមានតម្លៃស្មើសូន្យ ប្រសិនបើប្រព័ន្ធគ្រប់គ្រងសីតុណ្ហភាពជួបប្រទះការធ្លាក់ចុះវ៉ុលអគ្គិសនី (Voltage drop) នៅថ្ងៃប្រតិបត្តិការជាក់ស្តែង។ CKM លុបបំបាត់ការពឹងផ្អែកលើសំណាង ដោយអនុវត្តពិធីសារគ្រប់គ្រងហានិភ័យ (Risk Management Protocols) យ៉ាងតឹងរ៉ឹង ដើម្បីធានាបាននូវស្ថិរភាពទិន្នផលដាច់ខាត ទោះបីជាស្ថិតក្នុងអថេរបរិស្ថានបែបណាក៏ដោយ។

## ១. សវនកម្មហេដ្ឋារចនាសម្ព័ន្ធទីតាំង៖ ប៉ារ៉ាម៉ែត្ររូបវន្ត និងអគ្គិសនី

ប្រតិបត្តិការខ្នាតធំមិនអាចពឹងផ្អែកលើការប៉ាន់ស្មានដោយភ្នែកទទេនោះទេ។
នៅពេលទីតាំងត្រូវបានជ្រើសរើស វិស្វកររបស់ CKM ធ្វើការសវនកម្មរចនាសម្ព័ន្ធ (Structural audit) ភ្លាមៗ។ គោលដៅគឺមិនមែនវាយតម្លៃសោភ័ណភាពទេ ប៉ុន្តែដើម្បីវាស់ស្ទង់ "សមត្ថភាពផ្ទុកថាមពលអគ្គិសនីអតិបរមា" សម្ពាធនៃប្រព័ន្ធផ្គត់ផ្គង់ទឹក និងគម្លាតលំហូរចរាចរណ៍ (Traffic flow mapping)។ ប្រសិនបើទិន្នន័យនៃទីតាំងបង្ហាញថា វាមិនអាចទ្រទ្រង់បន្ទុករបស់ម៉ាស៊ីនរក្សាសីតុណ្ហភាពពាណិជ្ជកម្មដោយសុវត្ថិភាពទេនោះ ប្រព័ន្ធម៉ាស៊ីនភ្លើងកម្រិតឧស្សាហកម្មឯករាជ្យ នឹងត្រូវបានដាក់បញ្ចូលទៅក្នុងម៉ាទ្រីសប្រតិបត្តិការដោយស្វ័យប្រវត្តិ ដើម្បីទប់ស្កាត់ការដួលរលំប្រព័ន្ធកម្តៅ។

## ២. ម៉ាទ្រីសពេលវេលាបញ្ជូន (Time-Axis Propulsion)៖ ការធ្វើសមកាលកម្មប្រតិបត្តិការ

ការដឹកជញ្ជូន និងការរៀបចំ គឺជាប្រតិបត្តិការដែលត្រូវបានគណនាថយក្រោយពីម៉ោងបម្រើ (T-Zero)។
CKM បដិសេធកាលវិភាគដែលពឹងផ្អែកលើពាក្យថា "ប្រហែល"។ ប្រតិបត្តិការរបស់យើងដំណើរការលើម៉ាទ្រីសពេលវេលាដ៏តឹងរ៉ឹង៖ T-6 (៦ម៉ោងមុន T-Zero) ក្រុមវិស្វកម្មដំឡើងប្រព័ន្ធទឹក អគ្គិសនី និងរចនាសម្ព័ន្ធផ្ទះបាយម៉ូឌុល; T-2 (២ម៉ោងមុន T-Zero) យានយន្តខ្សែសង្វាក់ត្រជាក់ (Cold-chain fleet) មកដល់ទីតាំងដើម្បីផ្ទេរប្រូតេអ៊ីនចូលទៅក្នុងប្រព័ន្ធត្រួតពិនិត្យសីតុណ្ហភាពសកម្ម។ ការធ្វើសមកាលកម្មនៃទិន្នន័យពេលវេលានេះ គឺជាពិធីសារតែមួយគត់ដែលធានាថា អេនត្រូពី (Entropy) មិនអាចកើនឡើងនៅក្នុងប្រតិបត្តិការឡើយ។

## ៣. ពិធីសារបម្រុង (Contingency Protocols)៖ ការលុបបំបាត់អថេរខាងក្រៅ

សេវាកម្មកម្រិតកំពូល គឺជាសមត្ថភាពក្នុងការរក្សាទិន្នផលឱ្យនៅថេរ ទោះបីជាមានការប្រែប្រួលប៉ារ៉ាម៉ែត្រខាងក្រៅក៏ដោយ។
ប្រព័ន្ធរបស់ CKM ត្រូវបានរចនាឡើងជាមួយនឹងកម្រិតអត់ធ្មត់ (Tolerance levels) ដ៏ខ្ពស់។ សម្រាប់ប្រតិបត្តិការក្រៅអាគារ ឧបករណ៍អគ្គិសនីទាំងអស់ដំណើរការក្រោមស្តង់ដារការពារសំណើម (IP Rating) កម្រិតខ្ពស់ ហើយរចនាសម្ព័ន្ធជម្រកបម្រុងត្រូវបានគណនាទុកជាមុន។ ក្នុងករណីមានគម្លាតនៃទិន្នន័យភ្ញៀវចូលរួម ប្រព័ន្ធទែរម៉ូឌីណាមិករបស់យើងបានបម្រុងទុកនូវវត្ថុធាតុដើមអសកម្មរួចជាស្រេច ដែលអាចធ្វើសកម្មភាពកែច្នៃកម្ដៅ (Thermal activation) ភ្លាមៗ ដើម្បីបំពេញតម្រូវការបន្ថែមដោយមិនធ្វើឱ្យប៉ះពាល់ដល់ស្ថិរភាពនៃខ្សែសង្វាក់បញ្ជូនចម្បង។

> សមត្ថភាពប្រតិបត្តិការខ្នាតធំ មិនមែនកើតចេញពីការព្យាយាមដោះស្រាយបញ្ហានៅនឹងកន្លែងនោះទេ ប៉ុន្តែជាលទ្ធផលនៃការគណនាហានិភ័យ និងការដាក់ពង្រាយប្រព័ន្ធការពារជាមុន ដើម្បីធានាថាបញ្ហាមិនអាចកើតមានតាំងពីដំបូង។

### ទិន្នន័យរចនាសម្ព័ន្ធហិរញ្ញវត្ថុ និងប្រតិបត្តិការអគ្គិសនី

**តើរចនាសម្ព័ន្ធលំហូរសាច់ប្រាក់ (Cash Flow Structure) ត្រូវបានបែងចែកយ៉ាងដូចម្តេច ដើម្បីធានាស្ថិរភាពនៃខ្សែសង្វាក់ផ្គត់ផ្គង់?**
ដើម្បីចាក់សោរប្រភពវត្ថុធាតុដើមនៅក្នុងទីផ្សារសកល និងផ្តួចផ្តើមការសវនកម្មទីតាំង CKM អនុវត្តពិធីសារហិរញ្ញវត្ថុផ្អែកលើដំណាក់កាលប្រតិបត្តិការ (Milestone-based financial protocol)៖ ដំណាក់កាលទី១ សម្រាប់ការធានាកាលវិភាគ និងបញ្ជាទិញវត្ថុធាតុដើមកម្រិតកំពូល; ដំណាក់កាលទី២ សម្រាប់ការកៀរគរធនធានមនុស្ស និងភស្តុភារកម្មមុនថ្ងៃកម្មវិធី; និងដំណាក់កាលចុងក្រោយ ត្រូវបានផាត់ទាត់បន្ទាប់ពីប្រតិបត្តិការបញ្ចប់ដោយជោគជ័យ និងទិន្នផលស្របតាមប៉ារ៉ាម៉ែត្រដែលបានកំណត់។ រចនាសម្ព័ន្ធនេះការពារភាពរអាក់រអួលនៃប្រព័ន្ធភស្តុភារកម្មខាងក្រោយ ដោយធានាថាគ្រប់ធនធានត្រូវបានធ្វើសកម្មភាពទាន់ពេលវេលា។

**ប្រសិនបើមានការដាច់ចរន្តអគ្គិសនីនៅទីតាំង តើប្រព័ន្ធមានប្រតិកម្មយ៉ាងណាដើម្បីរក្សាសីតុណ្ហភាពអាហារ?**
ប្រតិបត្តិការរបស់ CKM មិនទទួលយកហានិភ័យនៃការរអាក់រអួលថាមពល (Power interruption) ឡើយ។ នៅក្នុងដំណាក់កាលសវនកម្ម ប្រសិនបើសមត្ថភាពអគ្គិសនីរបស់ទីតាំងមិនស្ថិតស្ថេរ ម៉ាស៊ីន phát điện (Generator) កម្រិតឧស្សាហកម្មនឹងត្រូវបានដំឡើងជាប្រព័ន្ធបម្រុង។ ប្រសិនបើបណ្តាញអគ្គិសនីចម្បងដួលរលំ ឧបករណ៍ប្តូរប្រភពថាមពលដោយស្វ័យប្រវត្តិ (Automatic Transfer Switch - ATS) នឹងភ្ជាប់ប្រព័ន្ធគ្រប់គ្រងសីតុណ្ហភាពស្នូល (Core thermal holding systems) ទៅកាន់ប្រភពថាមពលបម្រុងក្នុងរយៈពេលគិតជាមិល្លីវិនាទី។ ការផ្លាស់ប្តូរនេះធានាថា ខ្សែសង្វាក់ទែរម៉ូឌីណាមិកនៃអាហារមិនទទួលរងនូវការធ្លាក់ចុះសីតុណ្ហភាពឡើយ។
